```<?php
error_reporting(E_ALL);

    $id = $_POST['id'];

    $kategori = $_POST['kategori'];

        }

    $sql = 'UPDATE data_barang SET ';
    $sql .= "nama = '{$nama}', kategori = '{$kategori}', ";
```
